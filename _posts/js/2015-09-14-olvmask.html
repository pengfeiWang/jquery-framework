遮罩, 提示信息

{% highlight javascript %} //显示 $.ui.olvMask(); $.ui.olvMask.show(); //隐藏 $.ui.olvMask.hide(); //自动隐藏 $.ui.olvMask({ autoHide: true // 或者数字, 最小2000 }); {% endhighlight %}
{% highlight javascript %} // 自动隐藏必须有文本, 或者开启 loading $.ui.olvMask({ msg:'自定义文本', autoHide: true }); {% endhighlight %}
{% highlight javascript %} $.ui.olvMask({ msg:'自定义文本' }); {% endhighlight %}
{% highlight javascript %} $.ui.olvMask({ loading: true }); {% endhighlight %}

其他
